desc emp;
select empno,ename,hiredate,sal from emp where ename like 'K%';
select * from emp where hiredate >'2000-01-01' and hiredate<'2000-12-31';
select * from emp where comm is not NULL;
select ename,deptno,sal from emp where sal>1500 and deptno like 30;
select * from emp where deptno like 30 order by sal desc;
select * from emp order by sal desc;
select * from emp order by deptno, sal desc;
select * from emp where deptno order by deptno desc,ename,sal desc;
select ename,sal,comm,(sal+sal*comm/100) from emp where comm is not null order by (sal*comm/100) desc; 
select ename,sal, sal*(13/100), deptno from emp;
select ename,sal,round((sal/(12*5)),0) from emp where deptno like 20; 
select ename,sal,round(sal*(15/100),1) from emp where sal>=1500 and sal<=3000;
select ename,sal,sal*(90/100) from emp;
select lower(substr(ename,1,3)) from emp where length(ename)>=6;
select avg(sal),max(sal),min(sal) from emp where deptno like 10;
select deptno,job,count(empno) from emp group by deptno,job;
select job,count(*) from emp group by job having count(*)>=4;
